JSON Mutation Hooks

Register with hook_json!(host, JsonMutateTarget::Variant, route_pattern, callback). Return JSON_MUTATE_MODIFIED after writing output with write_json_output, or JSON_MUTATE_UNCHANGED to pass through.

Action object shape

{
  "id": "check_update",
  "label": "Check for updates",
  "step": "GET /api/system/update"
}
TargetSummaryInput JSONRoute matching
json.responseMutate the full JSON response object before it is sent.The serialized JSON response body as an object.Prefix match on the request path. Use `/api/system` or `*` for all JSON routes.
json.actionsMutate only the top-level `actions` array on API responses.A JSON array of action objects with `id`, `label`, and optional `step` fields.Prefix match on the request path. Runs after body hooks; changes are merged back into the response.

Examples

json.response

Mutate the full JSON response object before it is sent.

Register

hook_json!(host, JsonMutateTarget::ResponseBody, "/api/system", on_system_body);

Handler

extern "C" fn on_system_body(ctx: *const JsonMutateContext) -> i32 {
    let ctx = unsafe { &*ctx };
    let input = unsafe { std::slice::from_raw_parts(ctx.json_in_ptr, ctx.json_in_len) };
    let Ok(mut value) = serde_json::from_slice::<serde_json::Value>(input) else {
        return JSON_MUTATE_UNCHANGED;
    };
    if let Some(map) = value.as_object_mut() {
        map.insert("my_plugin".into(), serde_json::json!(true));
    }
    let Ok(out) = serde_json::to_vec(&value) else {
        return JSON_MUTATE_UNCHANGED;
    };
    write_json_output(ctx, &out)
}

json.actions

Mutate only the top-level `actions` array on API responses.

Register

hook_json!(host, JsonMutateTarget::ResponseActions, "/api/system", on_system_actions);

Handler

extern "C" fn on_system_actions(ctx: *const JsonMutateContext) -> i32 {
    let ctx = unsafe { &*ctx };
    let input = unsafe { std::slice::from_raw_parts(ctx.json_in_ptr, ctx.json_in_len) };
    let Ok(mut actions) = serde_json::from_slice::<Vec<serde_json::Value>>(input) else {
        return JSON_MUTATE_UNCHANGED;
    };
    actions.retain(|a| a.get("id").and_then(|v| v.as_str()) != Some("diagnostics"));
    actions.push(serde_json::json!({
        "id": "custom_step",
        "label": "Open plugin docs",
        "step": "GET /docs/plugins"
    }));
    let Ok(out) = serde_json::to_vec(&actions) else {
        return JSON_MUTATE_UNCHANGED;
    };
    write_json_output(ctx, &out)
}
